2021-02-09
LeetCode

移除链表元素

删除链表中等于给定值 val 的所有节点。

示例:

输入: 1->2->6->3->4->5->6, val = 6
输出: 1->2->3->4->5

来源：力扣（LeetCode）
链接：https://leetcode-cn.com/problems/design-linked-list

C

运行时间：16 ms
内存消耗：7.7 MB

直接使用原来的链表来进行移除节点操作

struct ListNode* removeElements(struct ListNode* head, int val){
    // 删除头节点
    while (head != NULL && head->val == val) {
        head = head->next;
    }

    // 删除非头节点
    struct ListNode *current = head;
    while (current != NULL) {
        struct ListNode *pNext = current->next;
        if (pNext != NULL && pNext->val == val) {
            current->next = pNext->next;
            free(pNext);
        } else {
            current = current->next;
        }
    }
    return head;
}

设置一个虚拟头结点在进行移除节点操作

struct ListNode* removeElements(struct ListNode* head, int val){
    struct ListNode* sentinel = (struct ListNode*)malloc(sizeof(struct ListNode));//申请虚拟头节点
     assert(sentinel != NULL);
    sentinel->val = 0;
    sentinel->next = NULL;   
    sentinel->next = head;

    struct ListNode *prev = sentinel, *curr = head, *toDelete = NULL;
    while (curr != NULL) 
    {
        if (curr->val == val)
        {
            prev->next = curr->next;
            toDelete = curr;
        } 
        else 
        {
            prev = curr;
        }  
        curr = curr->next;
        if (toDelete != NULL)
        {
            free(toDelete);
            toDelete = NULL;
        }
    }

    struct ListNode *ret = sentinel->next;
    free(sentinel);
    return ret;
}

Python3

运行时间：68 ms
内存消耗：17.7 MB

class Solution:
    def removeElements(self, head: ListNode, val: int) -> ListNode:
        pre = ListNode(0)
        pre.next = head
        while pre.next != None:
            if pre.next.val == val:
                if pre.next == head:
                    head = head.next
                    pre.next = pre.next.next
                else:
                    pre.next = pre.next.next
            else:
                pre = pre.next
        return head
}
